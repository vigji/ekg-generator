import numpy as np
from PIL import Image
import imageio.v2 as imageio

# carica immagine
img = Image.open("mindray.png").convert("RGB")
arr = np.array(img)
h, w, _ = arr.shape

# regioni delle curve (adattate al layout Mindray che hai fornito)
ecg_rect = (
    int(h*0.08),
    int(h*0.28),
    int(w*0.02),
    int(w*0.72),
)

spo_rect = (
    int(h*0.33),
    int(h*0.52),
    int(w*0.02),
    int(w*0.72),
)

def scroll_band(frame, rect, dx):
    y0,y1,x0,x1 = rect
    band = frame[y0:y1, x0:x1]
    W = band.shape[1]
    off = dx % W
    band2 = np.concatenate([band, band], axis=1)
    frame[y0:y1, x0:x1] = band2[:, off:off+W]
    return frame

# parametri video
fps = 25
duration = 10
frames = fps * duration
speed = 2

writer = imageio.get_writer(
    "mindray_monitor.mp4",
    fps=fps,
    codec="libx264",
    quality=8
)

for i in range(frames):
    frame = arr.copy()
    frame = scroll_band(frame, ecg_rect, i*speed)
    frame = scroll_band(frame, spo_rect, i*speed)
    writer.append_data(frame)

writer.close()

print("Video creato: mindray_monitor.mp4")

import numpy as np
import cv2

out_path = "Mindray_PM_VT_170bpm_loop.mp4"

W, H = 800, 400
plot_w = int(W * 0.7)
right_x = plot_w

green = (120,255,0)
cyan = (255,200,0)
red = (60,60,255)
gray = (160,160,160)

fps = 20
px_per_frame = 4
px_per_sec = fps * px_per_frame
hr = 170
rr_px = int(px_per_sec * 60 / hr)

beats = 8
buffer_w = beats * rr_px
signal = np.zeros(buffer_w)

qrs_width = max(3, int(px_per_sec * 0.18))

for beat in range(beats):
    base = beat * rr_px
    for i in range(qrs_width):
        idx = base + i
        if idx < buffer_w:
            phase = i / qrs_width
            signal[idx] += (
                1.5*np.exp(-((phase-0.35)/0.22)**2)
                -0.7*np.exp(-((phase-0.75)/0.18)**2)
            )

signal -= np.mean(signal)
signal /= np.max(np.abs(signal))

ecg_center = int(H*0.25)
scale = 110
ecg_y = (ecg_center - signal*scale).astype(np.int32)

fourcc = cv2.VideoWriter_fourcc(*'mp4v')
video = cv2.VideoWriter(out_path, fourcc, fps, (W,H))

num_frames = buffer_w // px_per_frame

for f in range(num_frames):
    offset = (f*px_per_frame) % buffer_w
    frame = np.zeros((H,W,3), dtype=np.uint8)

    for x in range(plot_w-1):
        i1 = (offset+x) % buffer_w
        i2 = (offset+x+1) % buffer_w
        cv2.line(frame,(x,ecg_y[i1]),(x+1,ecg_y[i2]),green,2,cv2.LINE_AA)

    cv2.putText(frame,"HR 170",(right_x+10,60),
                cv2.FONT_HERSHEY_SIMPLEX,0.6,green,2)

    cv2.putText(frame,"SpO2 ---",(right_x+10,200),
                cv2.FONT_HERSHEY_SIMPLEX,0.6,cyan,2)

    cv2.putText(frame,"---/---",(right_x+10,340),
                cv2.FONT_HERSHEY_SIMPLEX,0.6,red,2)

    video.write(frame)

video.release()

import numpy as np
from PIL import Image, ImageDraw

# OUTPUT
out_path = "mindray_vt_monomorphic_180bpm.gif"

# CANVAS
W, H = 800, 400
plot_w = int(W * 0.70)
right_x = plot_w
black = (0, 0, 0)
green = (0, 255, 120)
cyan  = (0, 200, 255)
red   = (255, 60, 60)

# ANIMATION / SWEEP (monitor-like)
fps = 25
px_per_frame = 4
px_per_sec = fps * px_per_frame

# VT @ 180 bpm
hr = 180
rr_px = int(px_per_sec * 60 / hr)

# Build monomorphic VT waveform (wide, stable)
beats = 10
buffer_w = beats * rr_px
sig = np.zeros(buffer_w, dtype=np.float32)

qrs_w = max(3, int(px_per_sec * 0.18))  # wide complex

for b in range(beats):
    base = b * rr_px
    for i in range(qrs_w):
        idx = base + i
        if idx >= buffer_w:
            break
        ph = i / qrs_w
        # asymmetric monomorphic VT morphology
        sig[idx] += (1.6 * np.exp(-((ph - 0.35) / 0.22) ** 2)
                     -0.8 * np.exp(-((ph - 0.75) / 0.18) ** 2))

# normalize
sig -= sig.mean()
sig /= np.max(np.abs(sig)) + 1e-9

ecg_center = int(H * 0.25)
scale = 110
ybuf = (ecg_center - sig * scale).astype(int)
x = np.arange(plot_w)

# frames: one seamless cycle
n_frames = buffer_w // px_per_frame
frames = []

for f in range(n_frames):
    off = (f * px_per_frame) % buffer_w
    y = np.roll(ybuf, -off)[:plot_w]
    pts = list(zip(x.tolist(), y.tolist()))

    img = Image.new("RGB", (W, H), black)
    d = ImageDraw.Draw(img)

    # ECG trace
    d.line(pts, fill=green, width=2)

    # Right column (Mindray-ish)
    d.text((right_x + 10, 40),  "HR",      fill=green)
    d.text((right_x + 10, 80),  "180",     fill=green)

    d.text((right_x + 10, 170), "SpO2",    fill=cyan)
    d.text((right_x + 10, 210), "---",     fill=cyan)

    d.text((right_x + 10, 290), "NIBP",    fill=red)
    d.text((right_x + 10, 330), "---/---", fill=red)

    frames.append(img)

frames[0].save(
    out_path,
    save_all=True,
    append_images=frames[1:],
    duration=int(1000 / fps),
    loop=0,
    optimize=False
)

print("Saved:", out_path)
